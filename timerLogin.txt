게임 타이머 정리

게임시작 클릭을 클릭하면

스테이지();
이게 돼야해

두더지대기 = null;
두더지등장 = null;

스테이지(){
    두더지대기 = setTimeout(() => {
        두더지나와();
        clearTimeout(두더지대기)
    }, 2000 ~ 5000(랜덤))
}

isClear = true;

두더지나와(){
    두더지 나오는 애니메이션

    두더지등장 = setTimeout(() => {
        isClear = false;
        clearTimeout(두더지등장);
        게임오버();
    }, 1000);
}

두더지 터치시(){
    if(스파이) 게임오버();
    else(){
        if(isClear === true){
            스테이지();
        }

    }
}

게임오버(){
    뭐 오버 하겠지
}
